⚙️ Jednorazowa konfiguracja (5 minut)

Dane będą zapisywane w Twoim Google Sheets — działa z każdego urządzenia bez logowania.

  1. Otwórz sheets.new → utwórz nowy arkusz → nadaj mu nazwę np. Mleko Sprzedaż
  2. W arkuszu kliknij menu Rozszerzenia → Apps Script
  3. Usuń domyślny kod i wklej poniższy skrypt, a potem kliknij 💾 Zapisz:
    const SHEET_NAME = 'Dane'; function doGet(e) { const p = e.parameter; const cb = p.callback; let result; if (p.action === 'add') { const sheet = getSheet(); const id = Date.now().toString(); sheet.appendRow([id, p.day, p.name || '', p.prod, Number(p.qty), new Date().toISOString()]); result = { ok: true, id }; } else if (p.action === 'delete') { const sheet = getSheet(); const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(p.id)) { sheet.deleteRow(i + 1); break; } } result = { ok: true }; } else if (p.action === 'clear') { const sheet = getSheet(); const last = sheet.getLastRow(); if (last > 1) sheet.deleteRows(2, last - 1); result = { ok: true }; } else { // domyślnie: pobierz dane const sheet = getSheet(); const rows = sheet.getDataRange().getValues(); result = rows.slice(1).map(r => ({ id: String(r[0]), day: r[1], name: r[2], prod: r[3], qty: Number(r[4]), ts: r[5] })).filter(r => r.id); } const output = cb ? cb + '(' + JSON.stringify(result) + ')' : JSON.stringify(result); return ContentService.createTextOutput(output) .setMimeType(cb ? ContentService.MimeType.JAVASCRIPT : ContentService.MimeType.JSON); } function getSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(['id','dzien','klient','produkt','ilosc','timestamp']); } return sheet; }
  4. Kliknij Wdróż → Nowe wdrożenie → Rodzaj: Aplikacja internetowa → Kto ma dostęp: Wszyscy → Wdróż → skopiuj URL
